    # ---------- HỌC NPC Tống Kim LƯU FILE theo campValue (qua phiên vẫn nhớ) ----------
    def _tk_load(self):
        try:
            import json, os
            if os.path.exists(self._tk_file):
                return json.load(open(self._tk_file, encoding="utf-8")) or {}
        except Exception: pass
        return {}

    def _tk_save_db(self):
        try:
            import json, os
            os.makedirs("data/output", exist_ok=True)
            json.dump(self._tk_db, open(self._tk_file, "w", encoding="utf-8"), ensure_ascii=False, indent=2)
        except Exception: pass

    def _tk_apply(self, dev, camp):
        """Nạp dữ liệu đã học của campValue này vào dict runtime per-dev (gọi lúc worker khởi động). DỨT KHOÁT:
        trường nào KHÔNG có record cho phe này thì XOÁ key per-dev → về fallback (tránh dùng id phe CŨ khi đổi phe)."""
        rec = self._tk_db.get(str(camp)) or {}
        for key, d in (("heal", self.tk_heal_npc), ("war", self.tk_war_npc),
                       ("war_pkt", self.tk_war_pkt), ("heal_pkt", self.tk_heal_pkt), ("heal_part", self.tk_heal_part)):
            if rec.get(key) is not None and rec.get(key) != "":
                v = rec[key]
                if key in ("heal", "war"):
                    try: v = int(v)   # id NPC: ép INT để khớp nids (int) ở at_camp — file lưu str → '23' in {ints} luôn False
                    except Exception: pass
                d[dev] = v
            else:
                d.pop(dev, None)
        self.tk_heal_slot.pop(dev, None)   # ô resolve lại mỗi phiên từ particular Ngũ Hoa
        if rec:
            self.log(f"📂 {dev}: nạp NPC TK đã học (camp {camp}): {rec}")

    def _tk_learn(self, dev, key, val):
        """Ghi 1 trường đã học vào DB theo campValue của dev + lưu file (nhớ qua phiên)."""
        camp = self.dev_camp.get(dev)
        if camp is None: return
        rec = self._tk_db.setdefault(str(camp), {})
        if rec.get(key) != val:
            rec[key] = val
            self._tk_save_db()

    def _tk_learn_map(self, dev, key, val):
        """Ghi id NPC học theo (campValue, MAP HIỆN TẠI) → maps[mapId][key]. Để 'cứ tới map đó + phe đó = mặc định
        gọi remote id này' (kể cả NPC ngoài tầm đọc). Lưu bền qua phiên. mapId lấy từ _cur_map (set ở _tk_autofind)."""
        camp = self.dev_camp.get(dev); mid = (getattr(self, '_cur_map', {}) or {}).get(dev)
        if camp is None or mid is None: return
        rec = self._tk_db.setdefault(str(camp), {}); maps = rec.setdefault("maps", {})
        mrec = maps.setdefault(str(mid), {})
        if mrec.get(key) != val:
            mrec[key] = val; self._tk_save_db()
            self.log(f"💾 {dev}: LƯU {key}={val} cho map {mid} phe {camp} (lần sau tự remote).")

    @staticmethod
    def _noacc(s):
        """Bỏ dấu + lower để khớp tên VN không lệ thuộc dấu (vd 'Phi Tốc Hoàn' → 'phi toc hoan')."""
        s = (s or "").lower()
        return "".join(c for c in unicodedata.normalize("NFD", s) if unicodedata.category(c) != "Mn")

    def _scan_lac(self, dev, b=None):
        """Sau khi mở cửa sổ: dump túi, phát hiện loại lắc nào CÓ THẬT (Phi Tốc/Lệnh Bài/Chiến Cổ) → union vào
        lac_available → select box chỉ hiện loại có. Chạy nền (dump_bag_items hơi nặng)."""
        b = b or self.bots.get(dev)
        if not b: return
        def _bg():
            try: items = (b.dump_bag_items() or {}).get("items") or []
            except Exception as e:
                self.log(f"🏃 {dev}: đọc túi lỗi {e}"); return
            bag = [it for it in items if it.get("location") == 2]
            found = []
            for lbl, pid, _g in LAC_TYPES:
                s = sum((it.get("stack") or 1) for it in bag if it.get("particular") == pid)
                self.lac_stock[pid] = s
                if s > 0: found.append(lbl)
            if found: self.lac_available |= set(found)
            self.log(f"🏃 {dev}: lắc có trong túi: {found or '(không có)'}")
            self.root.after(0, self._refresh_lac_boxes)
            self.root.after(0, self._refresh_lac_counts)
        threading.Thread(target=_bg, daemon=True).start()

    def _refresh_lac_boxes(self):
        """Đổ lại giá trị 3 select box = '—' + các loại lắc đang có trong túi. Lựa cũ không còn → về '—'."""
        vals = [LAC_NONE] + [lbl for lbl, _p, _g in LAC_TYPES if lbl in self.lac_available]
        for var, cb in zip(self.lac_sel, getattr(self, "lac_cbs", [])):
            try:
                cb["values"] = vals
                if var.get() not in vals: var.set(LAC_NONE)
            except Exception: pass

    def _refresh_lac_counts(self):
        """Label 'còn N' sau mỗi select box = số lượng loại đang chọn (từ lần dump túi gần nhất + tự trừ khi dùng)."""
        pid_by = {lbl: pid for lbl, pid, _g in LAC_TYPES}
        for var, cnt in zip(self.lac_sel, getattr(self, "lac_count_lbls", [])):
            try:
                pid = pid_by.get(var.get())
                n = self.lac_stock.get(pid) if pid is not None else None
                cnt.config(text=(f"còn {n}" if n is not None else ""))
            except Exception: pass

    def _active_lac(self):
        """Trả LIST (nhãn, particular, gap) các loại lắc đang được chọn ở 3 select box (bỏ '—', bỏ trùng)."""
        try: sels = {v.get() for v in self.lac_sel}
        except Exception: sels = set()
        return [(lbl, pid, gap) for lbl, pid, gap in LAC_TYPES if lbl in sels]

    def _fmt_player(self, fl):
        """Định dạng 1 người quét được: tên - phái(Tống/Kim·hệ) - cấp - %máu."""
        nm = fl.get("id.nameValue") or "?"
        cv = fl.get("id.campValue")
        camp = self._CAMP_LABEL.get(cv, f"c{cv}" if cv is not None else "?")
        he = self._SERIES_LABEL.get(fl.get("data.series"), "?")
        lv = fl.get("data.level")
        hp = fl.get("id.healthCurrent"); hm = fl.get("id.healthMax")
        try:
            pct = f"{int(round(hp / hm * 100))}%" if hp is not None and hm else (f"{hp}" if hp is not None else "?")
        except Exception:
            pct = "?"
        return f"{nm}-{camp}·{he}-Lv{lv if lv is not None else '?'}-{pct}"

    def _tk_autofind(self, dev, b, npcs, map_id):
        """MAP NÀO CŨNG GỌI ĐƯỢC NPC heal/war (id đổi theo map):
          (1) NHỚ THEO MAP: rec['maps'][mapId]={heal,war} → map đã đụng → gọi THẲNG (khỏi dò).
          (2) DÒ DIALOG: map lạ → mở từng NPC (op33), dialog có 'dược phẩm miễn phí' = HEAL, 'ra chiến trường' = WAR.
              Chỉ op33 (KHÔNG eNpcSelect = không lỡ kích hành động). Probe 1 lần/map (throttle 30s), cần ≥3 NPC."""
        camp = self.dev_camp.get(dev)
        if camp is None or map_id is None: return
        rec = self._tk_db.setdefault(str(camp), {}); maps = rec.setdefault("maps", {}); mkey = str(map_id)
        present = {str(n.get("id")) for n in npcs if str(n.get("id", "")).isdigit()}
        mrec = maps.get(mkey) or {}
        self._cur_map = getattr(self, '_cur_map', {}); self._cur_map[dev] = map_id   # cho _learn_npc lưu theo map
        import unicodedata as _ud
        def _na(_s):
            _s = (_s or "").lower()
            return "".join(_c for _c in _ud.normalize("NFD", _s) if _ud.category(_c) != "Mn")
        # MẶC ĐỊNH dùng id NPC đã HỌC theo (phe, MAP) — REMOTE (talk_npc/op33 KHÔNG check khoảng cách). User chốt:
        #   "cứ tới map đó, phe đó = mặc định gọi remote id này" (Trinh Sát ngoài tầm đọc vẫn gọi được). KHÔNG đòi
        #   present nữa. Name-match (dưới) VẪN chạy (throttle) để CẬP NHẬT nếu NPC hiện ra id KHÁC = đổi trận → re-learn.
        if mrec.get("heal"): self.tk_heal_npc[dev] = int(mrec["heal"])
        if mrec.get("war"):  self.tk_war_npc[dev] = int(mrec["war"])
        # ⚠️ NPC id ĐỔI MỖI TRẬN TỐNG KIM (14h: heal=24/war=23 → 15h: heal=2594/war=2582). id nạp từ file
        #    (top-level, trận TRƯỚC) thường KHÔNG còn quanh → CHỈ early-return khi heal+war đã set *VÀ ĐANG HIỆN
        #    DIỆN* quanh (đúng trận này). KHÔNG hiện diện = STALE → PHẢI name-match lại. (Bug LinhLinh đứng im:
        #    file heal='23'/war='2582' cũ → early-return cũ bỏ qua name-match → at_camp luôn False, không nhận máu/ra trận.)
        _h = self.tk_heal_npc.get(dev); _w = self.tk_war_npc.get(dev)
        if _h and _w and str(_h) in present and str(_w) in present:
            return   # đã đủ heal+war VÀ đúng trận hiện tại → khỏi name-match
        # ⭐ KHỚP THEO TÊN — AN TOÀN: ĐÃ BỎ dò-dialog op33 (op33-spam mở nhầm RƯƠNG CHỨA ĐỒ + gây CRASH "script
        #    destroyed"). TÊN NPC nằm ở Controller.data.name → DÙNG getNearNpcsDetail (getNearNpcsSafe đọc surface
        #    field = RỖNG). 'Quân Nhu'=nhận máu(heal); 'Trinh Sát'=ra trận(war). Đúng cả Tống/Kim, bỏ dấu. Throttle 5s.
        now = time.time(); pk = (dev, mkey)
        # GIẢM FREEZE: getNearNpcsDetail (đọc data.name NPC gần) ở TK đông lính spawn/chết = dễ chạm object freed
        #   → đơ game. ĐÃ BIẾT cả heal+war (set, dù Trinh Sát ngoài tầm) → quét THƯA 60s (chỉ để bắt đổi-trận id
        #   mới); CHƯA biết → quét dày 5s để học nhanh. Cắt phần lớn số lần đọc near-npc khi đã ổn định.
        # _known = đã biết heal+war VÀ ĐANG HIỆN DIỆN (tới đây = đã qua early-return, nên thực chất STALE) → name-match
        #   NHANH (5s) để bắt id trận mới, KHÔNG để 60s (id global stale = đứng im không ra trận, lỗi 3 cửa Kim).
        #   Freeze của getNearNpcsDetail giờ có LƯỚI A che (Process.setExceptionHandler).
        _known = bool(_h and _w and str(_h) in present and str(_w) in present)
        _gap = 60 if _known else 5
        if now - self._tk_probe_ts.get(pk, 0) < _gap:
            return
        self._tk_probe_ts[pk] = now
        try:
            named = (b.rpc.get_near_npcs_detail() or {}).get("npcs") or []
        except Exception:
            named = []
        ch = False
        for _n in named:
            _nid = _n.get("id"); _nm = _na(_n.get("name", ""))
            if not str(_nid).isdigit():
                continue
            # 🚫 LỌC BỎ "Chiêu binh quân" (chỗ báo danh) — cả Tống lẫn Kim ra-trận đều = 'Trinh Sát'. (user chốt)
            if "chieu binh" in _nm:
                continue
            if "quan nhu" in _nm and self.tk_heal_npc.get(dev) != int(_nid):
                self.tk_heal_npc[dev] = int(_nid); mrec["heal"] = int(_nid); ch = True
                if rec.get("heal_opt") is None: rec["heal_opt"] = 0
                self.log(f"❤️ {dev}: NPC máu theo TÊN = {_nid} ('{_n.get('name')}')")
            elif "trinh sat" in _nm and self.tk_war_npc.get(dev) != int(_nid):
                self.tk_war_npc[dev] = int(_nid); mrec["war"] = int(_nid); ch = True
                self.log(f"⚔️ {dev}: NPC ra trận theo TÊN = {_nid} ('{_n.get('name')}')")
        if ch:
            maps[mkey] = mrec; self._tk_save_db()
        if not (self.tk_heal_npc.get(dev) and self.tk_war_npc.get(dev)):
            if now - self._tk_nomatch_ts.get(pk, 0) > 60:   # throttle 60s (đỡ spam khi ngoài trại)
                self._tk_nomatch_ts[pk] = now
                _nm_list = [(n.get("id"), n.get("name")) for n in named if (n.get("name") or "").strip()][:16]

    # ---------- WATCHDOG: game crash/dis → user mở lại game → TỰ attach + chạy lại ----------
    def _watchdog_tick(self):
        """Cửa sổ mà USER muốn chạy (opened=True) NHƯNG vòng auto đã DỪNG (run=False) = đã crash/dis (vòng _loop
        tự set run=False khi poll_recv lỗi 8 lần). Đóng tay thì opened=False nên KHÔNG đụng vào. → thử kết nối lại
        (game mở lại = PID mới, connect() tự lấy PID mới; game chưa vào thế giới thì connect() trả False, thử lại sau)."""
        now = time.time()
        for dev in [d for d in list(self.opened) if self.opened.get(d)]:
            b = self.bots.get(dev)
            # CRASH = phiên Frida đã DETACH (game đóng/crash). KHÔNG dựa vào run nữa: giờ run=False là BÌNH THƯỜNG
            #   khi cửa sổ mới mở (chưa tick Ra trận). Chỉ recover khi bot CÓ rồi mà mất kết nối.
            if b is None:
                continue
            try: alive = b.is_connected()
            except Exception: alive = False
            if alive:                                   # còn kết nối → khoẻ → bỏ qua
                continue
            if self._recovering.get(dev):               # đang thử rồi → khỏi chồng
                continue
            if now - self._recover_ts.get(dev, 0) < 15:  # throttle: 15s/lần/cửa sổ (đỡ spam adb khi game chưa mở lại)
                continue
            self._recover_ts[dev] = now
            self._recovering[dev] = True
            threading.Thread(target=self._recover, args=(dev,), daemon=True).start()
        self.root.after(6000, self._watchdog_tick)

    def _recover(self, dev):
        try:
            b = self.bots.get(dev)
            if b is None:                               # chưa có bot (lạ) → mở mới như tick tay
                self.log(f"♻ {dev}: chưa có phiên → mở mới…")
                self._open(dev); return
            self._logn(f"recon:{dev}", f"♻ {dev}: phát hiện cửa sổ DỪNG (crash/dis?) → thử kết nối lại…", gap=60)
            try: ok = b.connect()                       # connect() tự lấy PID MỚI sau khi game relaunch
            except Exception as e:
                self.log(f"⏳ {dev}: reconnect lỗi {e} — thử lại sau"); return
            if not ok:                                  # game chưa chạy / chưa vào nhân vật → watchdog thử lại sau
                self._logn(f"reconwait:{dev}",
                           f"⏳ {dev}: chưa kết nối lại được (game chưa vào thế giới?) — sẽ thử lại", gap=60)
                return
            b.capture_sent = True                       # giữ học thao tác NPC sau reconnect
            for _ in range(4):
                try: b.poll_recv()
                except Exception: pass
                time.sleep(0.2)
            try: b.redetect_fd_by_traffic()
            except Exception: pass
            self.root.after(0, lambda: self.tree.set(dev, "auto", "✔"))
            self._read_bg(dev)
            self._scan_lac(dev, b)
            if self.fight.get():                        # CHỈ chạy lại vòng ra-trận nếu đang tick 'Ra trận'
                self.log(f"✅ {dev}: KẾT NỐI LẠI OK → chạy lại auto ra trận")
                self._fight_start(dev)
            else:
                self.log(f"✅ {dev}: KẾT NỐI LẠI OK (chưa tick Ra trận → chỉ giữ kết nối)")
        finally:
            self._recovering[dev] = False

    @staticmethod
    def _dialog_options(hexstr):
        """Trả LIST các đoạn chữ UTF-8 (>=4 ký tự) trong gói thoại — mỗi đoạn ~ 1 option/dòng."""
        try: raw = bytes.fromhex(hexstr)
        except Exception: return []
        out = []; cur = bytearray()
        for c in raw:
            if 0x20 <= c < 0x7f or c >= 0xc2: cur.append(c)
            else:
                if len(cur) >= 4:
                    try:
                        t = cur.decode("utf-8", "ignore").strip()
                        if len(t) >= 3: out.append(t)
                    except Exception: pass
                cur = bytearray()
        if len(cur) >= 4:
            try:
                t = cur.decode("utf-8", "ignore").strip()
                if len(t) >= 3: out.append(t)
            except Exception: pass
        return out

    @staticmethod
    def _readable(hexstr):
        """Trích các đoạn chữ UTF-8 đọc được từ hex gói (để xem option thoại NPC)."""
        try:
            raw = bytes.fromhex(hexstr)
        except Exception:
            return ""
        out = []; cur = bytearray()
        i = 0
        while i < len(raw):
            c = raw[i]
            if 0x20 <= c < 0x7f or c >= 0xc2:   # ascii in được hoặc đầu chuỗi UTF-8
                cur.append(c)
            else:
                if len(cur) >= 3:
                    try: out.append(cur.decode("utf-8", "ignore"))
                    except Exception: pass
                cur = bytearray()
            i += 1
        if len(cur) >= 3:
            try: out.append(cur.decode("utf-8", "ignore"))
            except Exception: pass
        return " | ".join(s.strip() for s in out if s.strip())

    @staticmethod
    def _dlg_opts(hx):
        """op34/124: trả LIST option (field protobuf 2, wire2 string) THEO THỨ TỰ.
        selectIndex của eNpcSelect = VỊ TRÍ trong list này (0-based; index 0 = eNpcSelect RỖNG)."""
        def vi(d, o):
            r = s = 0
            while o < len(d):
                x = d[o]; o += 1; r |= (x & 0x7F) << s
                if not x & 0x80: break
                s += 7
            return r, o
        out = []
        try: b = bytes.fromhex(hx)[6:]
        except Exception: return out
        o = 0
        while o < len(b):
            try: tag, o = vi(b, o)
            except Exception: break
            f, w = tag >> 3, tag & 7
            if w == 0: _, o = vi(b, o)
            elif w == 2:
                ln, o = vi(b, o); raw = b[o:o + ln]; o += ln
                if f == 2:
                    try: out.append("".join(c for c in raw.decode("utf-8", "ignore") if c.isprintable()))
                    except Exception: pass
            elif w == 1: o += 8
            elif w == 5: o += 4
            else: break
        return out

    def _opts_after_talk(self, b, npc_id):
        """Mở thoại NPC rồi đọc danh sách option (chuỗi) từ gói trả về (op34/33)."""
        try: seen0 = len(getattr(b, "recv_buffer", []) or [])
        except Exception: seen0 = 0
        try: b.talk_npc(str(npc_id))
        except Exception: pass
        opts = []
        for _ in range(5):                       # chờ + poll tới khi có gói thoại
            time.sleep(0.18)
            try: b.poll_recv()
            except Exception: pass
            try:
                for pk in (list(getattr(b, "recv_buffer", []) or []))[seen0:]:
                    if pk.get("opcode") in (34, 33):
                        for s in self._dialog_options(pk.get("hex") or ""):
                            if s not in opts: opts.append(s)
            except Exception: pass
            if opts: break
        return opts

    def _do_rejoin(self, b, dev):
        """Về doanh trại: nhận máu (Quan Nhu) rồi RA CHIẾN TRƯỜNG. Option NHẬN MÁU = HỌC từ cú click tay của
        bạn (nếu chưa học → KHÔNG bấm bừa, nhắc bạn tự click 1 lần). Cờ _rejoining để khỏi tự-học gói của bot."""
        self._rejoining[dev] = True
        heal_npc = self.tk_heal_npc.get(dev, TK_HEAL_NPC)   # id HỌC theo phe; fallback phe Tống
        war_npc  = self.tk_war_npc.get(dev, TK_WAR_NPC)
        try:
            # NHẬN DƯỢC PHẨM MIỄN PHÍ = mở Quan Nhu (op33) rồi gửi op232 eClientCompleted (rỗng) — bắt live từ click user
            # TỰ HỌC thuốc máu TK: dump túi TRƯỚC, sau khi nhận dump LẠI → particular nào tăng = "Ngũ Hoa Ngọc Lộ Hoàn".
            def _g6_counts():
                try:
                    its = (b.dump_bag_items() or {}).get('items') or []
                    c = {}
                    for it in its:
                        if it.get('genre') == 6 and it.get('location') == 2:
                            p = it.get('particular'); c[p] = c.get(p, 0) + (it.get('stack') or 1)
                    return c
                except Exception: return {}
            before = _g6_counts()
            # NHẬN THUỐC = mở NPC (op33) rồi:
            #   - Tống: CHỌN 1 LỰA CHỌN trong popup → eNpcSelect(option đã học từ click tay / file).
            #   - Kim : eClientCompleted (op232).
            # Ưu tiên option đã học (npc_learn theo click, hoặc heal_opt trong file); chưa học → thử op232 (Kim).
            camp = self.dev_camp.get(dev)
            heal_opt = self.npc_learn.get((dev, heal_npc))
            if heal_opt is None:
                heal_opt = (self._tk_db.get(str(camp)) or {}).get('heal_opt')
            if heal_opt is None:
                heal_opt = 0   # MẶC ĐỊNH: chọn DÒNG ĐẦU 'Dược phẩm miễn phí' = eNpcSelect RỖNG (đúng cho cả Tống/Kim
                               # NPC có option nhận thuốc; KHÔNG fallback op232 vì op232 SAI cho NPC này = 'túi KHÔNG đổi').
            try: _seen0 = len(getattr(b, "recv_buffer", []) or [])
            except Exception: _seen0 = 0
            b.talk_npc(str(heal_npc)); time.sleep(0.35)
            try: b.poll_recv()
            except Exception: pass
            # ✅ CHỌN INDEX ĐÚNG = ĐỌC MENU op34 (đã về từ 1 op33 ở trên, KHÔNG spam) → tìm dòng chứa từ-khoá nhận
            #   thuốc ('dược phẩm miễn phí'/'nhận thuốc'/'hồi phục') → eNpcSelect(index DÒNG ĐÓ). Index 'Dược phẩm
            #   miễn phí' KHÔNG luôn = 0 (tuỳ NPC/phe) → hardcode 0 = nhận hụt (bug 'chọn index sai'). Bỏ dấu khi khớp.
            import unicodedata as _ud
            def _na(_s):
                _s = (_s or "").lower()
                return "".join(_c for _c in _ud.normalize("NFD", _s) if _ud.category(_c) != "Mn")
            menu_idx = None
            try:
                for pk in (list(getattr(b, "recv_buffer", []) or []))[_seen0:]:
                    if pk.get("opcode") == 34:
                        opts = self._dlg_opts(pk.get("hex") or "")
                        for _i, _s in enumerate(opts):
                            _sn = _na(_s)
                            if any(_na(_k) in _sn for _k in self._HEAL_KW):
                                menu_idx = _i; break
                    if menu_idx is not None: break
            except Exception: menu_idx = None
            if menu_idx is not None:
                b.send_gs('eNpcSelect', selectIndex=int(menu_idx)); heal_opt = int(menu_idx)
            elif str(heal_opt) == 'op232':
                b.send_gs('eClientCompleted')
            elif heal_opt and int(heal_opt) != 0:
                b.send_gs('eNpcSelect', selectIndex=int(heal_opt))
            else:
                # KHÔNG đọc được menu → fallback index 0 TƯỜNG MINH (field '08 00', không gửi rỗng). Retry op232 ở dưới.
                b.send_gs('eNpcSelect', selectIndex=0)
            time.sleep(0.6)
            after = _g6_counts()
            gained = [(p, after[p] - before.get(p, 0)) for p in after if after[p] > before.get(p, 0)]
            # ✅ XÁC NHẬN (không fix mù): so túi TRƯỚC/SAU. Tăng = option nhận thuốc ĐÚNG; không đổi = SAI/chưa nhận.
            if gained:
                self.log(f"✅ {dev}: XÁC NHẬN NHẬN THUỐC OK — túi tăng {gained} (option {heal_opt} đúng)")
            else:
                # PHƯƠNG PHÁP CHÍNH TRƯỢT → THỬ PHƯƠNG PHÁP CÒN LẠI 1 LẦN (heal interface = war interface; war chạy
                #   bằng eNpcSelect index 0). Đối xứng: op232 trượt → thử eNpcSelect(0); eNpcSelect(0) trượt → thử
                #   op232. Cái nào ăn thì HỌC LẠI (chống 'op232' học cũ làm hỏng heal vĩnh viễn — bug live LinhDL).
                #   CHỈ 1 retry (KHÔNG brute-force op33 = crash 'script has been destroyed').
                try:
                    b.talk_npc(str(heal_npc)); time.sleep(0.35); b.poll_recv()
                    if str(heal_opt) == 'op232':
                        b.send_gs('eNpcSelect', selectIndex=0); alt = 0          # op232 hỏng → về eNpcSelect index 0 (như war)
                    else:
                        b.send_gs('eClientCompleted'); alt = 'op232'             # eNpcSelect hỏng → thử op232 (vài NPC Kim)
                    time.sleep(0.6)
                    _a2 = _g6_counts()
                    gained = [(p, _a2[p] - before.get(p, 0)) for p in _a2 if _a2[p] > before.get(p, 0)]
                except Exception:
                    gained = []; alt = None
                if gained:
                    self._tk_learn(dev, 'heal_opt', alt)
                    self.log(f"✅ {dev}: NHẬN THUỐC OK qua {alt} — túi tăng {gained}. ĐÃ HỌC LẠI heal_opt={alt}.")
                else:
                    self._logn(f"healfail:{dev}",
                               f"⚠️ {dev}: NPC {heal_npc} chưa ra thuốc (eNpcSelect-0 lẫn op232 trượt) — "
                               f"transient (chuyển map / túi đầy). Vòng sau thử lại.", gap=30)
            gained = [g[0] for g in gained]
            part = gained[0] if gained else self.tk_heal_part.get(dev)
            if part is not None:

    @staticmethod
    def _parse_skillid(hexstr):
        """op238/239/240: body bắt đầu ở byte 6, field1 (skillid) = tag 0x08 + varint."""
        try:
            b = bytes.fromhex(hexstr)
            if len(b) < 8 or b[6] != 0x08: return None
            val = 0; shift = 0; i = 7
            while i < len(b):
                byte = b[i]; val |= (byte & 0x7f) << shift; i += 1
                if not (byte & 0x80): break
                shift += 7
            return val or None
        except Exception:
            return None

    @staticmethod
    def _parse_op35(hexstr):
        """op35 eNpcSelect: field1 selectIndex (varint, tag 0x08). Cho phép 0."""
        try:
            b = bytes.fromhex(hexstr)
            if len(b) < 7 or b[6] != 0x08: return None
            val = 0; shift = 0; i = 7
            while i < len(b):
                c = b[i]; val |= (c & 0x7f) << shift; i += 1
                if not (c & 0x80): break
                shift += 7
            return val
        except Exception:
            return None

    @staticmethod
    def _parse_op33_npc(hexstr):
        """op33 eNpcDialogue: field1 npcId = chuỗi ASCII (tag 0x0a)."""
        try:
            b = bytes.fromhex(hexstr)
            if len(b) < 8 or b[6] != 0x0a: return None
            ln = b[7]; s = b[8:8 + ln].decode("ascii", "ignore")
            return s if s.isdigit() else None
        except Exception:
            return None

    def toggle_fight(self):
        """Tick 'Tự đánh người' = cho TẤT CẢ cửa sổ đã mở bắt đầu đánh; bỏ tick = dừng hết."""
        if self.fight.get():
